Welcome to payments!

LIST all payments: /payments [GET]
CREATE a payment: /payments [POST]
RETRIEVE a payment: /payments/{id} [GET]
UPDATE a payment: /payments/{id} [PUT]
DELETE a payment: /payments/{id} [DELETE]
QUERY payments: /payments?{key}={value} [GET]
SET default payment: /payments/{id}/set-default [PUT]
CHARGE default payment: /payments/charge [PUT]
UPDATE a payment: /payments/{id} [PATCH]
